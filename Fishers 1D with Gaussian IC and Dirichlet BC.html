    # Working ok
    import numpy as np
    from scipy.integrate import solve_ivp
    from numpy.linalg import norm
    from scipy.optimize import minimize
    import matplotlib.pyplot as plt
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # length of the rod
    N = 1000 # number of grid points
    dx = L / (N - 1)  # grid spacing in x direction
    # Define the initial condition
    def initial_condition(x_val):
        mu=L/2; # mean
        sigma=0.1 # Standard deviation
        return np.exp(-(x_val - mu)**2 / (2 * sigma**2)) # Gaussian Distribution
    # Define the function to solve
    def f(t, u,r_value):
        rhs = np.zeros_like(u)
        for i in range(1, N-1):
            uxx = (u[i+1] - 2*u[i] + u[i-1]) / dx**2
            rhs[i] = D*uxx + r_value*u[i]*(1-u[i])
        # Apply Neumann boundary conditions
        #mu = L/2  # mean
        #sigma = 0.1  # Standard deviation
        u[0] = u[1]
        u[-1] = u[-2]
        # Dirichlets BC not good now.
        #u[0] = np.exp(-(mu**2) / (2 * sigma**2)) * np.sin((np.pi/2) * (1 - t))
        #u[-1] = np.exp((1 - mu**2) / (2 * sigma**2)) * np.sin((np.pi/2) * t)
        #u[0] = u[1] - (dx * np.exp(-(mu**2) / (2 * sigma**2)) * np.sin((np.pi/2) * (1 - t)))
        #u[-1] = u[-2] + (dx * np.exp((1 - mu**2) / (2 * sigma**2)) * np.sin((np.pi/2) * (1-t)))
        
        return rhs
    # Define the grid of points
    x = np.linspace(0, L, N)
    # Set the initial condition
    u0 = initial_condition(x)
    # List to store the solutions
    solutions = []
    # Range of r values
    r_min = 6
    r_max = 15
    num_r_values = 100
    r_values = np.linspace(r_min, r_max, num_r_values)  # 100 r_values are from 6 to 15
    # Solve the ODE system for different r values
    for r_value in r_values:
        # Solve the ODE system for r_value
        # We pass extra argument args=(r_value,) which is a tuple and function must be defined as a lambda function
        # It does not work for function f (needs a lambda function) and calculates the solutions when t=1 only
        # Here I have used Backward Difference Formula (BDF) which is faster than RK method.
        
        sol = solve_ivp(lambda t, u: f(t, u, r_value), [0, 1], u0, method='BDF', args=(r_value,), t_eval=[1])
        # Save the solution at t=1 for the current r_value
        solutions.append(sol.y[:, -1])
    index_r10 = np.where(r_values == 10)[0][0]
    save_r_val = solutions[index_r10] # Solution set when r=10
    alpha = 0.01
    # Define the Tikhnov function
    def tikhnov(r_value):
        # In the Tikhnov Function We use the closest value to r_value (which is generated from the minimize function)
        # from the list r_values, here norm give L2 norm by default. 
        norm_L2 = norm(save_r_val - solutions[np.argmin(np.abs(r_values - r_value))])**2 + alpha * r_value**2
        return norm_L2
    initial_guess = 9.89
    # Find the value of r that minimizes the Tikhnov function
    iterations = 0
    r_val_iterations = []
    def callback(x):
        global iterations
        global r_val_iterations
        iterations += 1
        r_val_iterations.append(x[0])
    # Find the value of r that minimizes the Tikhnov function
    res = minimize(tikhnov, x0=np.array([initial_guess]), method='Nelder-Mead', callback=callback, options={'disp': True})
    r_opt = res.x[0]
    # Print the optimal value of r
    print('............')
    print("Optimal value of r:", r_opt)
    print('')
    # Print all the r_values in each iteration
    r_val_iterations = np.array(r_val_iterations)
    print('The r values of each iteration: \n', r_val_iterations)
    # Solve the ODE system for r=10 and r_opt
    sol_r_10 = solve_ivp(lambda t, u: f(t, u, 10), [0, 1], u0, method='BDF', args=(10,))
    sol_r_opt = solve_ivp(lambda t, u: f(t, u, r_opt), [0, 1], u0, method='BDF', args=(r_opt,))
    # Plot the solutions
    plt.plot(x, sol_r_10.y[:, -1], label='r = 10')
    plt.plot(x, sol_r_opt.y[:, -1], label='r = {}'.format(r_opt))
    plt.xlabel('x')
    plt.ylabel('u')
    plt.title('Solutions at t = 1')
    plt.legend()
    plt.grid(True)
    plt.show()
    print('........')
    # Plot the iteration number against the corresponding r values
    plt.plot(range(1, iterations + 1), r_val_iterations, marker='o')
    plt.xlabel('Iteration')
    plt.ylabel('r value')
    plt.title('Iteration vs. r value')
    plt.grid(True)
    plt.show()
