    # Fisher's 1D - Methods of Lines
    import numpy as np
    import matplotlib.pyplot as plt
    from scipy.integrate import solve_ivp
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # length of the rod
    N = 100  # number of grid points
    dx = L / (N - 1)  # grid spacing in x direction
    r = 10 
    ​
    # Define the initial condition
    def initial_condition(x_val):
        return np.sin(np.pi*x_val)
    ​
    def f(t, u):
        rhs = np.zeros_like(u)
        for i in range(1, N-1):
            uxx = (u[i+1] - 2*u[i] + u[i-1]) / dx**2
            rhs[i] = D*uxx + r*u[i]*(1-u[i])
        # Apply boundary condition u(0) = u(L) = 0
        u[0] = 0
        u[-1] = 0
        return rhs
    ​
    ​
    # Define the grid of points
    x = np.linspace(0, L, N)
    ​
    # Set the initial condition
    u0 = initial_condition(x)
    ​
    # Solve the ODE system
    sol = solve_ivp(f, [0, 1], u0, method='RK23', t_eval=np.linspace(0, 1, 1001))
    ​
    # Plot the solution
    plt.plot(x, sol.y[:, 0], label='t=0')
    plt.plot(x, sol.y[:, 250], label='t=0.25')
    plt.plot(x, sol.y[:, 500], label='t=0.5')
    plt.plot(x, sol.y[:, 750], label='t=0.75')
    plt.plot(x, sol.y[:, 1000], label='t=1')
    plt.xlabel('x')
    plt.ylabel('Temperature')
    plt.legend()
    plt.show()
    ​
    # Temperature at x=L/2 as a function of time
    plt.figure(figsize=(10, 5))
    index_L2 = int((N-1)/2)  # index of x=L/2
    plt.plot(sol.t, sol.y[index_L2, :])
    plt.xlabel('Time')
    plt.ylabel('Temperature at x=L/2')
    plt.show()

    # 2D format with optimized and 3D plot
    import numpy as np
    from scipy.integrate import solve_ivp
    import matplotlib.pyplot as plt
    from mpl_toolkits.mplot3d import Axes3D
    ​
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # Length of the domain in x direction
    W = 1.0  # Length of the domain in y direction
    Nx = 100  # Number of grid points in x direction
    Ny = 100  # Number of grid points in y direction
    dx = L / (Nx - 1)  # Grid spacing in x direction
    dy = W / (Ny - 1)  # Grid spacing in y direction
    r = 10
    ​
    # Define the initial condition
    def initial_condition(x, y):
        return np.sin(np.pi * x) * np.sin(np.pi * y)
    ​
    # Define the PDE system
    def f(t, u):
        u = u.reshape((Nx, Ny)) # we pass an 1D array as u but need to reshape for uxx and uyy
        rhs = np.zeros_like(u)
    ​
        uxx = (u[2:, 1:-1] - 2*u[1:-1, 1:-1] + u[:-2, 1:-1]) / dx**2
        uyy = (u[1:-1, 2:] - 2*u[1:-1, 1:-1] + u[1:-1, :-2]) / dy**2
    ​
        rhs[1:-1, 1:-1] = D * (uxx + uyy) + r * u[1:-1, 1:-1] * (1 - u[1:-1, 1:-1])
    ​
        # Apply boundary conditions
        rhs[0, :] = 0
        rhs[-1, :] = 0
        rhs[:, 0] = 0
        rhs[:, -1] = 0
    ​
        return rhs.ravel() # Again convert to a 1D array as the return value
    ​
    # Define the grid of points
    x = np.linspace(0, L, Nx)
    y = np.linspace(0, W, Ny)
    X, Y = np.meshgrid(x, y)
    ​
    # Set the initial condition
    u0 = initial_condition(X, Y).flatten()  # Initial Temperature of all the Grid points using sin function
    ​
    # Solve the PDE system
    sol = solve_ivp(f, [0, 1], u0, method='RK45', t_eval=np.linspace(0, 1, 15)) # method can be 'BDF' and only 15 times as it takes time.
    ​
    # Reshape the solution to plot
    U = sol.y.reshape((Nx, Ny,-1)) # (100,100,15)
     
    # Create a 3D plot
    fig = plt.figure(figsize=(10, 6))
    ax=plt.axes(projection='3d')
    ​
    # Plot the 3D surface
    ax.plot_surface(X, Y, U[:, :, 0], cmap='viridis') # U[:,:,0] is the temp(sol) of x and y points when t=0
    ax.set_xlabel('x')
    ax.set_ylabel('y')
    ax.set_zlabel('Temperature')
    ax.set_title('Temperature Evolution')
    plt.show()

    # Working ok
    import numpy as np
    from scipy.integrate import solve_ivp
    from numpy.linalg import norm
    from scipy.optimize import minimize
    import matplotlib.pyplot as plt
    ​
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # length of the rod
    N = 50  # number of grid points
    dx = L / (N - 1)  # grid spacing in x direction
    ​
    # Define the initial condition
    def initial_condition(x_val):
        return np.sin(np.pi * x_val)
    ​
    def f(t, u, r_value):
        rhs = np.zeros_like(u)
        for i in range(1, N - 1):
            uxx = (u[i + 1] - 2 * u[i] + u[i - 1]) / dx**2
            rhs[i] = D * uxx + r_value * u[i] * (1 - u[i])
        # Apply boundary condition u(0) = u(L) = 0
        u[0] = 0
        u[-1] = 0
        return rhs
    ​
    # Define the grid of points
    x = np.linspace(0, L, N)
    ​
    # Set the initial condition
    u0 = initial_condition(x)
    ​
    # List to store the solutions
    solutions = []
    ​
    # Range of r values
    r_min = 6
    r_max = 15
    num_r_values = 100
    r_values = np.linspace(r_min, r_max, num_r_values)  # 100 r_values are from 6 to 15
    ​
    # Solve the ODE system for different r values
    for r_value in r_values:
        # Solve the ODE system for r_value
        # We pass extra argument args=(r_value,) which is a tuple and function must be defined as a lambda function
        # It does not work for function f (needs a lambda function) and calculates the solutions when t=1 only
        # Here I have used Backward Difference Formula (BDF) which is faster than RK method.
        
        sol = solve_ivp(lambda t, u: f(t, u, r_value), [0, 1], u0, method='BDF', args=(r_value,), t_eval=[1])
    ​
        # Save the solution at t=1 for the current r_value
        solutions.append(sol.y[:, -1])
    ​
    index_r10 = np.where(r_values == 10)[0][0]
    save_r_val = solutions[index_r10] # Solution set when r=10
    ​
    alpha = 0.00001
    ​
    # Define the Tikhnov function
    def tikhnov(r_value):
        # In the Tikhnov Function We use the closest value to r_value (which is generated from the minimize function)
        # from the list r_values, here norm give L2 norm by default. 
        norm_L2 = norm(save_r_val - solutions[np.argmin(np.abs(r_values - r_value))])**2 + alpha * r_value**2
        return norm_L2
    ​
    initial_guess = 18.63
    ​
    # Find the value of r that minimizes the Tikhnov function
    iterations = 0
    r_val_iterations = []
    ​
    def callback(x):
        global iterations
        global r_val_iterations
        iterations += 1
        r_val_iterations.append(x[0])
    # Find the value of r that minimizes the Tikhnov function
    res = minimize(tikhnov, x0=np.array([initial_guess]), method='Nelder-Mead', callback=callback, options={'disp': True})
    r_opt = res.x[0]
    ​
    # Print the optimal value of r
    print('............')
    print("Optimal value of r:", r_opt)
    print('')
    ​
    # Print all the r_values in each iteration
    r_val_iterations = np.array(r_val_iterations)
    print('The r values of each iteration: \n', r_val_iterations)
    ​
    # Solve the ODE system for r=10 and r_opt
    sol_r_10 = solve_ivp(lambda t, u: f(t, u, 10), [0, 1], u0, method='BDF', args=(10,))
    sol_r_opt = solve_ivp(lambda t, u: f(t, u, r_opt), [0, 1], u0, method='BDF', args=(r_opt,))
    ​
    # Plot the solutions
    plt.plot(x, sol_r_10.y[:, -1], label='r = 10')
    plt.plot(x, sol_r_opt.y[:, -1], label='r = {}'.format(r_opt))
    plt.xlabel('x')
    plt.ylabel('u')
    plt.title('Solutions at t = 1')
    plt.legend()
    plt.grid(True)
    plt.show()

    plt.plot(range(1, iterations + 1), r_val_iterations, marker='o')
    plt.xlabel('Iteration')
    plt.ylabel('r value')
    plt.title('Iteration vs. r value')
    plt.grid(True)
