    import numpy as np
    from scipy.integrate import solve_ivp
    import matplotlib.pyplot as plt
    from mpl_toolkits.mplot3d import Axes3D
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # Length of the domain in x direction
    W = 1.0  # Length of the domain in y direction
    Nx = 100  # Number of grid points in x direction
    Ny = 100  # Number of grid points in y direction
    dx = L / (Nx - 1)  # Grid spacing in x direction
    dy = W / (Ny - 1)  # Grid spacing in y direction
    r = 10
    # Define the initial condition
    def initial_condition(x_val,y_val):
        mu=L/2; # mean
        sigma=0.1 # Standard deviation
        return np.exp(-(x_val - mu)**2 / (2 * sigma**2))* np.exp(-(y_val - mu)**2 / (2 * sigma**2)) # Gaussian Distribution
    # Define the PDE system
    def f(t, u):
        u = u.reshape((Nx, Ny)) # we pass an 1D array as u but need to reshape for uxx and uyy
        rhs = np.zeros_like(u)
        uxx = (u[2:, 1:-1] - 2*u[1:-1, 1:-1] + u[:-2, 1:-1]) / dx**2
        uyy = (u[1:-1, 2:] - 2*u[1:-1, 1:-1] + u[1:-1, :-2]) / dy**2
        rhs[1:-1, 1:-1] = D * (uxx + uyy) + r * u[1:-1, 1:-1] * (1 - u[1:-1, 1:-1])
        # Apply Neumann boundary conditions du/dy=0
        u[0, :] = u[1,:]
        u[-1, :] = u[-2,:]
        # Apply Neumann boundary conditions du/dx= 0
        u[:, 0] = u[:,1]
        u[:, -1] = u[:,-2]
        return rhs.ravel() # Again convert to a 1D array as the return value
    # Define the grid of points
    x = np.linspace(0, L, Nx)
    y = np.linspace(0, W, Ny)
    X, Y = np.meshgrid(x, y)
    # Set the initial condition
    u0 = initial_condition(X, Y).flatten()  # Initial Temperature of all the Grid points using sin function
    # Solve the PDE system
    sol = solve_ivp(f, [0, 1], u0, method='RK45', t_eval=np.linspace(0, 1, 15)) # method can be 'BDF' and only 15 times as it takes time.
    # Reshape the solution to plot
    U = sol.y.reshape((Nx, Ny,-1)) # (100,100,15)
     
    # Create a 3D plot
    fig = plt.figure(figsize=(10, 6))
    ax=plt.axes(projection='3d')
    # Plot the 3D surface
    ax.plot_surface(X, Y, U[:, :, 0], cmap='viridis') # U[:,:,0] is the temp(sol) of x and y points when t=0
    ax.set_xlabel('x')
    ax.set_ylabel('y')
    ax.set_zlabel('Temperature')
    ax.set_title('Temperature Evolution')
    plt.show()

    # 2D format Tikhonov Regularization
    import numpy as np
    from scipy.integrate import solve_ivp
    import matplotlib.pyplot as plt
    from mpl_toolkits.mplot3d import Axes3D
    from scipy.optimize import minimize
    from numpy.linalg import norm
    # Define the parameters
    D = 1.0  # Diffusion Coefficient
    L = 1.0  # Length of the domain in x direction
    W = 1.0  # Length of the domain in y direction
    Nx = 25  # Number of grid points in x direction
    Ny = 25  # Number of grid points in y direction
    dx = L / (Nx - 1)  # Grid spacing in x direction
    dy = W / (Ny - 1)  # Grid spacing in y direction
    r = 10
    def initial_condition(x_val,y_val):
        mu=L/2; # mean
        sigma=0.1 # Standard deviation
        return np.exp(-(x_val - mu)**2 / (2 * sigma**2))* np.exp(-(y_val - mu)**2 / (2 * sigma**2)) # Gaussian Distribution
    # List to store the solutions
    solutions = []
    # Range of r values
    r_min = 6
    r_max = 15
    num_r_values = 100
    r_values = np.linspace(r_min, r_max, num_r_values)  # 100 r_values are from 6 to 15
    # Define the PDE system
    def f(t, u, r_value):
        u = u.reshape((Nx, Ny)) # we pass an 1D array as u but need to reshape for uxx and uyy
        rhs = np.zeros_like(u)
        uxx = (u[2:, 1:-1] - 2*u[1:-1, 1:-1] + u[:-2, 1:-1]) / dx**2
        uyy = (u[1:-1, 2:] - 2*u[1:-1, 1:-1] + u[1:-1, :-2]) / dy**2
        rhs[1:-1, 1:-1] = D * (uxx + uyy) + r_value * u[1:-1, 1:-1] * (1 - u[1:-1, 1:-1])
        # Apply Neumann boundary conditions du/dy=0
        u[0, :] = u[1,:]
        u[-1, :] = u[-2,:]
        # Apply Neumann boundary conditions du/dx= 0
        u[:, 0] = u[:,1]
        u[:, -1] = u[:,-2]
        return rhs.ravel() # Again convert to a 1D array as the return value
    # Define the grid of points
    x = np.linspace(0, L, Nx)
    y = np.linspace(0, W, Ny)
    X, Y = np.meshgrid(x, y)
    # Set the initial condition
    u0 = initial_condition(X, Y).flatten()  # Initial Temperature of all the Grid points using sin function
    for r_value in r_values:
        # Solve the PDE system
        sol = solve_ivp(lambda t,u: f(t,u,r_value), [0, 1], u0, method='RK45', args=((r_value,)),t_eval=[1]) # method can be RK45 and t=1.
        solutions.append(sol.y)
        
    index_r10 = np.where(r_values == 10)[0][0]
    save_r_val = solutions[index_r10].flatten() # Solution set when r=10 and need to be 1D array so flatten
    alpha = 0.000000001
    # Define the Tikhnov function
    def tikhnov(r_value):
        # In the Tikhnov Function We use the closest value to r_value (which is generated from the minimize function)
        # from the list r_values, here norm give L2 norm by default. 
        global solutions
        save_r_val = solutions[index_r10].flatten() # Solution set when r=10 and need to be 1D array so flatten
        solutions=np.array(solutions).reshape(len(r_values),Nx*Ny)
        norm_L2 = norm(save_r_val - solutions[np.argmin(np.abs(r_values - r_value))])**2 + alpha * r_value**2
        return norm_L2
    initial_guess = 10.86
    # Find the value of r that minimizes the Tikhnov function
    iterations = 0
    r_val_iterations = []
    def callback(x):
        global iterations
        global r_val_iterations
        iterations += 1
        r_val_iterations.append(x[0])
        print("Iteration:", iterations, "r_value:", x[0])
        
    # Find the value of r that minimizes the Tikhnov function
    res = minimize(tikhnov, x0=np.array([initial_guess]), method='Nelder-Mead', callback=callback, options={'disp': True})
    r_opt = res.x[0]
    # Solve the ODE system for r=10 and r_opt at multiple time points
    sol_r_10 = solve_ivp(lambda t, u: f(t, u, 10), [0, 1], u0, method='RK45', t_eval=[1])
    sol_r_opt = solve_ivp(lambda t, u: f(t, u, r_opt), [0, 1], u0, method='RK45', t_eval=[1])
    # Reshape the solutions to plot
    U_r_10 = np.array(sol_r_10.y).reshape((Nx, Ny, -1))
    U_r_opt = np.array(sol_r_opt.y).reshape((Nx, Ny, -1))
    # Create a 3D plot with two subplots
    fig = plt.figure(figsize=(12, 6))
    ax1 = fig.add_subplot(121, projection='3d')
    ax2 = fig.add_subplot(122, projection='3d')
    # Plot the 3D surface for r=10
    ax1.plot_surface(X, Y, U_r_10[:, :, 0], cmap='viridis')
    ax1.set_xlabel('x')
    ax1.set_ylabel('y')
    ax1.set_zlabel('Temperature')
    ax1.set_title('Temperature Evolution for r=10')
    # Plot the 3D surface for r=r_opt
    ax2.plot_surface(X, Y, U_r_opt[:, :, 0], cmap='viridis')
    ax2.set_xlabel('x')
    ax2.set_ylabel('y')
    ax2.set_zlabel('Temperature')
    ax2.set_title(label='Temperature Evaluation for r = {}'.format(r_opt))
    plt.show()
    print('........')
    # Plot the iteration number against the corresponding r values
    plt.plot(range(1, iterations + 1), r_val_iterations, marker='o')
    plt.xlabel('Iteration')
    plt.ylabel('r value')
    plt.title('Iteration vs. r value')
    plt.grid(True)
    plt.show()
